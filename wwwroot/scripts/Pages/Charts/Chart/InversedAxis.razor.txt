<div class="control-section">
    <EjsChart id="inversedaxis" title="Exchange Rate(INR per USD)" Width="60%">
        <ChartArea><ChartAreaBorder Width="0"></ChartAreaBorder></ChartArea>
        <ChartPrimaryXAxis valueType="Syncfusion.EJ2.Blazor.Charts.ValueType.Category" opposedPosition="true" isInversed="true">
            <ChartPrimaryXAxisMajorGridLines Width="0"></ChartPrimaryXAxisMajorGridLines>
        </ChartPrimaryXAxis>
        <ChartPrimaryYAxis LabelStyle="@labelstyle" isInversed="true" EdgeLabelPlacement="EdgeLabelPlacement.Shift" LabelIntersectAction="LabelIntersectAction.Rotate45">

            <ChartPrimaryYAxisMajorTickLines Width="0"></ChartPrimaryYAxisMajorTickLines>
