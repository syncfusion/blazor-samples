    public void OnHeaderPositionChange(Syncfusion.EJ2.Blazor.DropDowns.ChangeEventArgs<string> args)
    {
        HeaderValue = (HeaderPosition)Enum.Parse(typeof(HeaderPosition), (args.Value as string));
    }
    public void OnHeaderStyleChange(Syncfusion.EJ2.Blazor.DropDowns.ChangeEventArgs<string> args)
    {
