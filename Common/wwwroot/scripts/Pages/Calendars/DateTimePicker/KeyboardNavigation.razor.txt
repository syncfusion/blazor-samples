<ActionDescription>
<p>
    DateTimePicker functionalities can be accessed interactively through keyboard shortcuts. Various key combinations enable efficient interaction with the calendar and time popup.
  </p>
  <p>
    <strong>Focus Navigation:</strong>
  </p>
  <ul>
    <li><span class="key-class"><kbd>Alt</kbd> + <kbd>J</kbd></span> — Focuses on the first component of the sample</li>
  </ul>
  <p>
    <strong>Input Navigation:</strong>
  </p>
  <ul>
    <li><span class="key-class"><kbd>Alt</kbd> + <kbd>Down Arrow</kbd></span> — Opens the popup</li>
    <li><span class="key-class"><kbd>Alt</kbd> + <kbd>Down Arrow</kbd></span> — Toggle between calendar and time selector popup when open</li>
    <li><span class="key-class"><kbd>Esc</kbd></span> — Closes the popup</li>
  </ul>
  <p>
    <strong>Date Selection:</strong>
  </p>
  <ul>
    <li><span class="key-class"><kbd>Enter</kbd></span> — Selects the currently focused date</li>
  </ul>
  <p>
    <strong>DatePicker Popup Navigation:</strong>
  </p>
  <ul>
    <li><span class="key-class"><kbd>Up Arrow</kbd></span> — Focuses the same day on the previous week</li>
    <li><span class="key-class"><kbd>Down Arrow</kbd></span> — Focuses the same day on the next week</li>
    <li><span class="key-class"><kbd>Left Arrow</kbd></span> — Focuses the previous day</li>
    <li><span class="key-class"><kbd>Right Arrow</kbd></span> — Focuses the next day</li>
    <li><span class="key-class"><kbd>Home</kbd></span> — Focuses the first day of the month</li>
    <li><span class="key-class"><kbd>End</kbd></span> — Focuses the last day of the month</li>
    <li><span class="key-class"><kbd>Page Up</kbd></span> — Focuses the same date on the previous month</li>
    <li><span class="key-class"><kbd>Page Down</kbd></span> — Focuses the same date on the next month</li>
    <li><span class="key-class"><kbd>Shift</kbd> + <kbd>Page Up</kbd></span> — Focuses the same date on the previous year</li>
    <li><span class="key-class"><kbd>Shift</kbd> + <kbd>Page Down</kbd></span> — Focuses the same date on the next year</li>
    <li><span class="key-class"><kbd>Ctrl</kbd> + <kbd>Home</kbd></span> — Focuses the first date of the current year</li>
    <li><span class="key-class"><kbd>Ctrl</kbd> + <kbd>End</kbd></span> — Focuses the last date of the current year</li>
    <li><span class="key-class"><kbd>Alt</kbd> + <kbd>Right Arrow</kbd></span> — Focuses throughout the popup container in forward direction</li>
    <li><span class="key-class"><kbd>Alt</kbd> + <kbd>Left Arrow</kbd></span> — Focuses throughout the popup container in backward direction</li>
  </ul>
  <p>
    <strong>DatePicker Popup View Navigation:</strong>
  </p>
  <ul>
    <li><span class="key-class"><kbd>Ctrl</kbd> + <kbd>Up Arrow</kbd></span> — Moves to inner view level (month to year, year to decade)</li>
    <li><span class="key-class"><kbd>Ctrl</kbd> + <kbd>Down Arrow</kbd></span> — Moves to outer view level (decade to year, year to month)</li>
  </ul>
  <p>
    <strong>Time Selection:</strong>
  </p>
  <ul>
    <li><span class="key-class"><kbd>Enter</kbd></span> — Selects the currently focused item and closes the popup</li>
  </ul>
  <p>
    <strong>TimePicker Navigation:</strong>
  </p>
  <ul>
    <li><span class="key-class"><kbd>Up Arrow</kbd></span> — Navigate and select the previous item</li>
    <li><span class="key-class"><kbd>Down Arrow</kbd></span> — Navigate and select the next item</li>
    <li><span class="key-class"><kbd>Home</kbd></span> — Navigate and select the first item</li>
    <li><span class="key-class"><kbd>End</kbd></span> — Navigate and select the last item</li>
  </ul>
</ActionDescription>


<div class="control-section col-lg-12">
    <div class="control-wrapper">
        <label class="example-label">Select Date and Time</label>
        <SfDateTimePicker TValue="DateTime?" @ref="@DateTimePickerObj"
                          Min='@MinDate' Max='@MaxDate' ShowClearButton="true" FullScreen="true" Placeholder="Select date and time" aria-label="Date and time picker with keyboard navigation">
            <DateTimePickerEvents TValue="DateTime?" OnRenderDayCell="@DisableDate"></DateTimePickerEvents>
        </SfDateTimePicker>
    </div>
</div>

@code {
    private SfDateTimePicker<DateTime?>? DateTimePickerObj { get; set; }
    public DateTime MinDate { get; set; } = new DateTime(2017, 08, 07, 00, 00, 00);
    public DateTime MaxDate { get; set; } = new DateTime(2024, 08, 27, 12, 00, 00); 
    
    private void DisableDate(RenderDayCellEventArgs args)
    {
        if (DateTimePickerObj != null)
        {
            var view = DateTimePickerObj.CurrentView();
            if (view == "Month" && (args.Date.Day == 10 || args.Date.Day == 15 || args.Date.Day == 25))
            {
                args.IsDisabled = true;
            }
        }
    }
}
