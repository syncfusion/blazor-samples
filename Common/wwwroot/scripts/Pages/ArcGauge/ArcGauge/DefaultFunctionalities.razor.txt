@code {
    public Theme Theme { get; set; }
    public string FontWeight = "500";
    public double YoutubeAngle = 338;
    public string YouTubeRadius = "105%";
    public double InstagramAngle = 331;
    public string InstagramRadius = "95%";
    public double TwitterAngle = 334;
    public string TwitterRadius = "76%";
    public double FacebookAngle = 321;
    public string FacebookRadius = "72%";
    public double TiktokAngle = 323;
    public string TiktokRadius = "52%";
    public string TikTokValueRadius = "24%";

    protected override void OnInitialized()
    {
        Theme = ThemeHelper.GetCurrentTheme(NavigationManager.Uri);
        if (Theme == Theme.Tailwind || Theme == Theme.TailwindDark)
        {
            YoutubeAngle = 336;
            InstagramAngle = 329;
            TwitterAngle = 332;
            FacebookAngle = 317;
            TiktokAngle = 318;
            InstagramRadius = "95%";
            TikTokValueRadius = "25%";
            TiktokRadius = "50%";
            FontWeight = "500";
        }
        else if (Theme == Theme.Bootstrap5 || Theme == Theme.Bootstrap5Dark) {
            FontWeight = "500";
        }
        else if (Theme == Theme.Fluent ||Theme == Theme.FluentDark) {
            FontWeight = "600";
        }
        else {
            FontWeight = "normal";
        }
    }
}
