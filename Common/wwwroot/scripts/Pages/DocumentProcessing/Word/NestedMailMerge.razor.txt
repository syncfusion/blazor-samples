@code {
    MemoryStream documentStream;
    string documentType = "DOCX";
    string Group1 = "Letter";
    string Group2 = "Implicit";
    //Hidden:Lines
    
    Dictionary<string, MemoryStream> fileData = new Dictionary<string, MemoryStream>();
    
    //End:Hidden
    /// <summary>
    /// Create and download the simple Word document
    /// </summary>
    public async Task GenerateDocument(MouseEventArgs args)
    {
        await LoadTemplate();
        NestedMailMergeService service = new NestedMailMergeService(fileData);
        documentStream = service.NestedMailMerge(Group1, Group2, documentType, "Generate Document");
        service.Close();
        if (documentType == "DOC")
            await JS.SaveAs("Sample.doc", documentStream.ToArray());
        else if (documentType == "WordML")
            await JS.SaveAs("Sample.xml", documentStream.ToArray());
        else
            await JS.SaveAs("Sample.docx", documentStream.ToArray());
    }
    /// <summary>
    /// Create and download the template Word document
    /// </summary>
    public async Task ViewTemplate(MouseEventArgs args)
    {
        await LoadTemplate();
        NestedMailMergeService service = new NestedMailMergeService(fileData);
        documentStream = service.NestedMailMerge(Group1, Group2, documentType, "View Template");
        service.Close();
        string outputFileName = "Template_Letter.doc";
        if (Group1 == "Report")
            outputFileName = "Template_Report.doc";
        await JS.SaveAs(outputFileName, documentStream.ToArray());
    }
    /// <summary>
    /// Load the template document
    /// </summary>
    public async Task LoadTemplate()
    {
        if (fileData != null && fileData.Count == 0)
        {
            string fileDataPath = NavigationManager.BaseUri + SampleService.WebAssetsPath;
            byte[] data = await Http.GetByteArrayAsync(fileDataPath + "data/word/template-report.doc");
            fileData.Add("template-report.doc", new MemoryStream(data));
            data = await Http.GetByteArrayAsync(fileDataPath + "data/word/template-letter.doc");
            fileData.Add("template-letter.doc", new MemoryStream(data));
            data = await Http.GetByteArrayAsync(fileDataPath + "data/word/employees.xml");
            fileData.Add("employees.xml", new MemoryStream(data));
        }
    }
}
