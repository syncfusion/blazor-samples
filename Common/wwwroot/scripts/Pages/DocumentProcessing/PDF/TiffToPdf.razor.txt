    /// <summary>
    /// Create and download the simple PDF document
    ///<summary>
    protected async void ConvertToPdf(MouseEventArgs args)
    {        
        Dictionary<string, MemoryStream> fileData = new Dictionary<string, MemoryStream>();
        string fileDataPath = NavigationManager.BaseUri + SampleService.WebAssetsPath;
        byte[] data = await Http.GetByteArrayAsync(fileDataPath + "data/pdf/image.tiff");
        fileData.Add("image.tiff", new MemoryStream(data));
        TiffToPdfService service = new TiffToPdfService(fileData);
        MemoryStream documentStream = service.ConvertToPdf();
        service.Close();
        await JS.SaveAs("TiffToPdf.pdf", documentStream.ToArray());
    }
    /// <summary>
    /// Download the template TIFF image used on this sample.
    /// </summary>
    /// <param name="args"></param>
    protected async void ViewTemplate(MouseEventArgs args)
    {
        string fileDataPath = NavigationManager.BaseUri + SampleService.WebAssetsPath;
        byte[] data = await Http.GetByteArrayAsync(fileDataPath + "data/pdf/image.tiff");
        await JS.SaveAs("image.tiff", data);
    }
}
