<ActionDescription>
<p>This example demonstrates the checkbox functionalities of MultiSelect component for intuitive multi-item selection.</p>

<p>Checkbox features:</p>
<ul>
  <li>Click MultiSelect element to open list</li>
  <li>Type character in search box to filter items</li>
  <li>Filtered list displays based on typed characters</li>
  <li>Select multiple values through checkboxes</li>
</ul>

<p>Built-in support configuration:</p>
<ul>
  <li>Enable checkbox support via <a href="https://help.syncfusion.com/cr/blazor/Syncfusion.Blazor.DropDowns.SfMultiSelect-2.html#Syncfusion_Blazor_DropDowns_SfMultiSelect_2_Mode" aria-label="Navigate to Mode property">Mode</a> property set to <code>CheckBox</code></li>
</ul>

<p>Additional options:</p>
<ul>
  <li><a href="https://help.syncfusion.com/cr/blazor/Syncfusion.Blazor.DropDowns.SfMultiSelect-2.html#Syncfusion_Blazor_DropDowns_SfMultiSelect_2_ShowSelectAll" aria-label="Navigate to ShowSelectAll property">Show SelectAll</a> – Show or hide the SelectAll option</li>
  <li><a href="https://help.syncfusion.com/cr/blazor/Syncfusion.Blazor.DropDowns.SfMultiSelect-2.html#Syncfusion_Blazor_DropDowns_SfMultiSelect_2_ShowDropDownIcon" aria-label="Navigate to ShowDropDownIcon property">Dropdown Button</a> – Show or hide the dropdown button</li>
  <li><a href="https://help.syncfusion.com/cr/blazor/Syncfusion.Blazor.DropDowns.SfMultiSelect-2.html#Syncfusion_Blazor_DropDowns_SfMultiSelect_2_EnableSelectionOrder" aria-label="Navigate to EnableSelectionReorder property">Selection Reorder</a> – Reorder selected items in popup visibility state</li>
</ul>

<p><b>See also</b></p>
<ul>
  <li><a href="https://blazor.syncfusion.com/documentation/multiselect-dropdown/checkbox/" target="_blank" rel="noopener" aria-label="Navigate to Blazor MultiSelect CheckBox documentation">Blazor MultiSelect - CheckBox</a></li>
</ul>
</ActionDescription>

    public bool ShowSelectAllCheckBox { get; set; } = true;
    public bool EnableSelectionOrders { get; set; } = true;
    public bool EnableDropDownIcon { get; set; } = true;

    private bool CheckSelectAll { get; set; } = true;
    private bool DropDownButton { get; set; } = true;
    private bool SelectionReorder { get; set; } = true;

    public void OnSelectAllChange(Syncfusion.Blazor.Buttons.ChangeEventArgs<bool> args)
    {
        this.ShowSelectAllCheckBox = args.Checked;
    }

    public void OnDropDownButtonChange(Syncfusion.Blazor.Buttons.ChangeEventArgs<bool> args)
    {
        this.EnableDropDownIcon = args.Checked;
    }

    public void OnSelectionReorder(Syncfusion.Blazor.Buttons.ChangeEventArgs<bool> args)
    {
        this.EnableSelectionOrders = args.Checked;
    }

    public class Countries
