  <li>Use SVG icons for visual representation</li>
  <li>Access data using implicit parameter <strong>context</strong> inside template</li>

<div class="col-lg-12 control-section">
    <div id="listbox-control">
        <SfListBox CssClass="e-template-listbox" TValue="string[]" DataSource="@AvailableItems" TItem="ItemData">
            <ListBoxFieldSettings Text="ItemName"></ListBoxFieldSettings>
            <ListBoxTemplates TItem="ItemData">
                <ItemTemplate>
                    <div class="list-wrapper">
                        <span class="@((context as ItemData).AvatarClass) e-avatar e-avatar-xlarge e-avatar-circle"></span>
                        <span class="text">@((context as ItemData).ItemName)</span>
                        <span class="description">@((context as ItemData).ItemDescription)</span>
                    </div>
                </ItemTemplate>
            </ListBoxTemplates>
        </SfListBox>
    </div>
</div>

@code {

    private List<ItemData> AvailableItems { get; set; } = new List<ItemData>
    {
        new ItemData { ItemName = "JavaScript", AvatarClass = "javascript", ItemDescription = "It is a lightweight interpreted or JIT-compiled programming language." },
        new ItemData { ItemName = "TypeScript", AvatarClass = "typescript", ItemDescription = "It is a typed superset of JavaScript that compiles to plain JavaScript." },
        new ItemData { ItemName = "Angular", AvatarClass = "angular", ItemDescription = "It is a TypeScript-based open-source web application framework." },
        new ItemData { ItemName = "React", AvatarClass = "react", ItemDescription = "A JavaScript library for building user interfaces. It can also render on the server using Node." },
        new ItemData { ItemName = "Vue", AvatarClass = "vue", ItemDescription = "A progressive framework for building user interfaces and it is incrementally adoptable." }
    };

    public class ItemData
    {
        public string? ItemName { get; set; }
        public string? AvatarClass { get; set; }
        public string? ItemDescription { get; set; }
    }

}
