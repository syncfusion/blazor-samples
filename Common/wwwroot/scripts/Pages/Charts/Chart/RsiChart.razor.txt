    private Theme Theme { get; set; }
    public string Width { get; set; } = "90%";
    private RSIChartData[] ChartPoints { get; set; } = new RSIChartData[] { };

    protected override async Task OnInitializedAsync()
    {
        ChartPoints = await Http.GetFromJsonAsync<RSIChartData[]>(NavigationManager.BaseUri + SampleService?.WebAssetsPath + "data/chart/financial-data.json") ?? null!;
    }

    protected override void OnInitialized()
    {
        
        Theme = ThemeHelper.GetCurrentTheme(NavigationManager.Uri);
        @*Hidden:Lines*@
        if (SampleService != null && SampleService.IsDevice)
        {
            Width = "100%";
        }
        @*End:Hidden*@
    }

    public class RSIChartData
    {
        public DateTime Period { get; set; }
        public double High { get; set; }
        public double Low { get; set; }
        public double Open { get; set; }
        public double Close { get; set; }
        public double Volume { get; set; }
    }
}
