@code {


    private Theme Theme { get; set; }
    public string Width { get; set; } = "90%";
    public class ChartData

        public double Revenue { get; set; }
    }

    public List<ChartData> GradientBarData = new List<ChartData>
    {
        new ChartData { Product = "Mobile",       Revenue = 260 },
        new ChartData { Product = "Integrations", Revenue = 310 },
        new ChartData { Product = "Support",      Revenue = 360 },
        new ChartData { Product = "Billing",      Revenue = 420 },
        new ChartData { Product = "CRM",          Revenue = 460 },
        new ChartData { Product = "Analytics",    Revenue = 520 }
    };

    protected override void OnInitialized()
    {
        Theme = ThemeHelper.GetCurrentTheme(NavigationManager.Uri);
        @*Hidden:Lines*@
        if (SampleService != null && SampleService.IsDevice)
        {
            Width = "100%";
        }
        @*End:Hidden*@
    }
}
