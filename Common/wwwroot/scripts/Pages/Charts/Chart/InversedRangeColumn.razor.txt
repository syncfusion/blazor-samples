@code {


    private Theme Theme { get; set; }
    public string Width { get; set; } = "90%";
    public class RangeData
    {
        public string Country { get; set; }
        public double Low { get; set; }
        public double High { get; set; }
    }

    private List<RangeData> rangeBarData = new List<RangeData>
    {
        new RangeData { Country = "Solomon Islands", Low = 44, High = 134 },
        new RangeData { Country = "Tonga", Low = 52, High = 131 },
        new RangeData { Country = "Trinidad and Tobago", Low = 36, High = 151 },
        new RangeData { Country = "Samoa", Low = 49, High = 131 },
        new RangeData { Country = "Saint Lucia", Low = 39, High = 148 },
        new RangeData { Country = "Georgia", Low = 68, High = 122 },
        new RangeData { Country = "Peru", Low = 56, High = 141 },
        new RangeData { Country = "Grenada", Low = 41, High = 147 },
        new RangeData { Country = "Dominica", Low = 46, High = 143 },
        new RangeData { Country = "Ukraine", Low = 64, High = 148 },
        new RangeData { Country = "Colombia", Low = 64, High = 134 }
    };

    protected override void OnInitialized()
    {
        
        Theme = ThemeHelper.GetCurrentTheme(NavigationManager.Uri);
        if (SampleService.IsDevice)
        {
            Width = "100%";
        }
    }
}
