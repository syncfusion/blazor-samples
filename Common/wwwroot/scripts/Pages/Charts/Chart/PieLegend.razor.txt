<ActionDescription>
<p>This sample displays internet usage statistics by country using a doughnut chart with legends positioned at the bottom for easy reference.</p>

<p>In this example, you can see how to render a doughnut chart with integrated legend functionality. Customize the doughnut appearance using the following properties:</p>
<ul>
  <li><a target='_blank' href='https://help.syncfusion.com/cr/blazor/Syncfusion.Blazor.Charts.AccumulationChartSeries.html#Syncfusion_Blazor_Charts_AccumulationChartSeries_Radius' aria-label="Navigate to the Radius property reference for accumulation chart series">Radius</a> - Controls the outer size of the doughnut</li>
  <li><a target='_blank' href='https://help.syncfusion.com/cr/blazor/Syncfusion.Blazor.Charts.AccumulationChartSeries.html#Syncfusion_Blazor_Charts_AccumulationChartSeries_InnerRadius' aria-label="Navigate to the InnerRadius property reference for accumulation chart series">InnerRadius</a> - Controls the inner hollow area size</li>
  <li>Legend text wrapping using the <code>TextWrap</code> property for improved readability</li>
</ul>

<p>More information about the legend feature can be found in this <a target='_blank' href='https://blazor.syncfusion.com/documentation/accumulation-chart/legend' aria-label="Navigate to the documentation for legend in accumulation chart component">documentation section</a>.</p>

    <SfAccumulationChart  EnableBorderOnMouseMove="false" SelectionMode="AccumulationSelectionMode.Point" Theme="@Theme">

@code{

    
    private Theme Theme { get; set; }
    public LegendPosition position { get; set; } = LegendPosition.Bottom;
    public List<PieData> PieChartPoints { get; set; } = new List<PieData>
    {
         new PieData { Country =  "China", Share = 35, DataLabelMappingName = "35%" },
         new PieData { Country =  "India", Share = 30, DataLabelMappingName = "30%" },
         new PieData { Country =  "USA", Share = 10.7, DataLabelMappingName = "10.7%" },
         new PieData { Country =  "Indonesia", Share = 7, DataLabelMappingName = "7%" },
         new PieData { Country =  "Brazil", Share = 5.3, DataLabelMappingName = "5.3%" },
         new PieData { Country =  "Others", Share = 12, DataLabelMappingName = "12%" }
    };

    protected override void OnInitialized()
    {
        
        Theme = ThemeHelper.GetCurrentTheme(NavigationManager.Uri);
    }

    public class PieData
    {
        public string? Country { get; set; }
        public double Share { get; set; }
        public string? DataLabelMappingName { get; set; }
    }
}
