<ActionDescription>
<p>This sample uses an Area series with an inversed axis to visualize stock prices from an alternative perspective. Crosshair provides time and value readouts for the selected position.</p>

<p>Configuration:</p>
<ul>
  <li>Invert an axis by setting the <a target="_blank" rel="noopener" href="https://help.syncfusion.com/cr/blazor/Syncfusion.Blazor.Charts.StockChartCommonAxis.html#Syncfusion_Blazor_Charts_StockChartCommonAxis_IsInversed" aria-label="Navigate to IsInversed property"><strong>IsInversed</strong></a> property to <strong>true</strong> on the common axis</li>
  <li>Axis inversion flips the scale direction to support specific analytical perspectives</li>
  <li>Useful for comparing upside-down patterns or alternative trend interpretations</li>
</ul>

<p>Interaction: Crosshair is enabled for detailed inspection. More on axis configuration: <a target="_blank" rel="noopener" href="https://blazor.syncfusion.com/documentation/stock-chart/axis-types" aria-label="Navigate to the documentation section">axis types documentation</a>.</p>

@if (StockChartData == null)

    <div class="stockchart-loader"></div>

            </StockChartSeriesCollection>

                <StockChartChartAreaBorder Width="0"></StockChartChartAreaBorder>
            </StockChartChartArea>
        </SfStockChart>
    </div>

@code
{
    private List<TechnicalIndicators> IndicatorList { get; set; } = new List<TechnicalIndicators>();
    private List<TrendlineTypes> TrendlineList { get; set; } = new List<TrendlineTypes>();
    private ChartData[]? StockChartData { get; set; }
    public string TranslateY { get; set; } = "-5px";
    
    protected override async Task OnInitializedAsync()
    {
        SetTheme();
        StockChartData = await Http.GetFromJsonAsync<ChartData[]>(NavigationManager.BaseUri + SampleService?.WebAssetsPath + "data/stock-chart/googl.json");
    }

    public class ChartData
    {
        public DateTime x { get; set; }
        public double open { get; set; }
        public double low { get; set; }
        public double close { get; set; }
        public double high { get; set; }
        public double volume { get; set; }
    }

    private string LoaderClass { get; set; } = "stockchart-loader";
    private string ChartContainerClass { get; set; } = "stockchart-container";

    private void OnChartLoaded(StockChartEventArgs args)
    {
        LoaderClass = string.Empty;
        ChartContainerClass = string.Empty;
        StateHasChanged();
    }

    private Theme CurrentTheme { get; set; } = Theme.Bootstrap4;

    private void SetTheme()
    {
        CurrentTheme = ThemeHelper.GetCurrentTheme(NavigationManager.Uri);
        if (CurrentTheme == Theme.Material3 || 
            CurrentTheme == Theme.Material3Dark || 
            CurrentTheme == Theme.Bootstrap4 || 
            CurrentTheme == Theme.Material || 
            CurrentTheme == Theme.MaterialDark)
        {
            TranslateY = "-7px";
        }
    }
}
