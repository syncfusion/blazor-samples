<ActionDescription>
<p>This sample demonstrates the navigation functionality of the <a href="https://www.syncfusion.com/blazor-components/blazor-breadcrumb" aria-label="Blazor Breadcrumb">Blazor Breadcrumb</a> component, enabling URL-based navigation by clicking breadcrumb items.</p>
    
    <p>Navigation features include:</p>
    
    <ul>
        <li>Automatic breadcrumb generation based on the current page's href (URL)</li>
        <li>Manual breadcrumb configuration using the <a href="https://help.syncfusion.com/cr/blazor/Syncfusion.Blazor.Navigations.SfBreadcrumb.html#Syncfusion_Blazor_Navigations_SfBreadcrumb_Url" aria-label="Url">Url</a> property when <code>BreadcrumbItems</code> is not specified</li>
        <li>URL navigation enabled for breadcrumb items</li>
        <li>Custom item rendering through the <a href="https://help.syncfusion.com/cr/blazor/Syncfusion.Blazor.Navigations.SfBreadcrumb.html#Syncfusion_Blazor_Navigations_SfBreadcrumb_ItemRendering" aria-label="ItemRendering">ItemRendering</a> event</li>
    </ul>
    
    <p><b>See also:</b></p>
    <ul>
        <li><a href="https://blazor.syncfusion.com/documentation/breadcrumb/breadcrumb-items" target="_blank" rel="noopener" aria-label="Blazor Breadcrumb - Items">Blazor Breadcrumb - Items</a></li>
    </ul>
</ActionDescription>

@code {
    private string BreadcrumbUrl { get; set; } = "https://blazor.syncfusion.com/demos/breadcrumb/bind-to-location?theme=bootstrap5";
    private string LocationBreadcrumbActiveItem { get; set; } = string.Empty;
    private string UrlBreadcrumbActiveItem { get; set; } = string.Empty;

// Customize breadcrumb item rendering
    private void OnBreadcrumbItemRendering(BreadcrumbItemRenderingEventArgs args)
    {
        if (args.Item?.Text == "demos")
        {
            args.Cancel = true;
        }
    }

// Reset breadcrumb state when the reset button is clicked
    private void ResetBreadcrumbState()
    {
        LocationBreadcrumbActiveItem = UrlBreadcrumbActiveItem = "";
    }

}
