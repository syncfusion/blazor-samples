<ActionDescription>
<p>This sample demonstrates the DataGrid selection feature for interactive row and cell selection.</p>
  
  <p>The Blazor DataGrid provides comprehensive selection support with simple mouse and keyboard interactions. To enable selection, set the <a target='_blank' href='https://help.syncfusion.com/cr/blazor/Syncfusion.Blazor.Grids.SfGrid-1.html#Syncfusion_Blazor_Grids_SfGrid_1_AllowSelection' aria-label="Navigate to the AllowSelection property reference for DataGrid component">AllowSelection</a> property to <strong>true</strong>.</p>
  
  <p><strong>Selection Types (via GridSelectionSettings):</strong></p>
  <ul>
    <li><strong>Single</strong> - Allows selection of one row or cell at a time (default)</li>
    <li><strong>Multiple</strong> - Allows selection of multiple rows or cells simultaneously</li>
  </ul>
  
  <p><strong>Selection Modes (via GridSelectionSettings):</strong></p>
  <ul>
    <li><strong>Row</strong> - Enables row selection (default)</li>
    <li><strong>Cell</strong> - Enables individual cell selection</li>
    <li><strong>Both</strong> - Enables both row and cell selection; clicking a cell selects both the row and cell</li>
  </ul>
  
  <p><strong>Keyboard Shortcuts for Multi-Selection:</strong></p>
  <ul>
    <li><strong>Ctrl + Click</strong> - Select multiple non-adjacent rows or cells</li>
    <li><strong>Shift + Click</strong> - Select a continuous range of rows or cells</li>
  </ul>
  
  <p>For more detailed information about the selection feature, refer to the <a target='_blank' href='https://blazor.syncfusion.com/documentation/datagrid/selection' aria-label="Navigate to the documentation for selection in DataGrid component">selection documentation</a>.</p>
</ActionDescription>

    public List<OrdersDetails>? GridData { get; set; }

    public class SelectionType
    {
        public string? id { get; set; }
        public string? type { get; set; }
    }

    private List<SelectionType> types = new List<SelectionType>()
    {
        new SelectionType(){ id= "Multiple", type= "Multiple" },
        new SelectionType(){ id= "Single", type= "Single" },
    };

    public class Mode
    {
        public string? id { get; set; }
        public string? mode { get; set; }
    }

    private List<Mode> modes = new List<Mode>()
    {
        new Mode(){ id= "Row", mode= "Row" },
        new Mode(){ id= "Cell", mode= "Cell" },
        new Mode(){ id= "Both", mode= "Both" },
    };

    protected override void OnInitialized()
    {
        
        GridData = OrdersDetails.GetAllRecords();
    }
