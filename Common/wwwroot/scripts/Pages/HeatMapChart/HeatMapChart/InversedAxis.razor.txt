<ActionDescription>
<p>This sample illustrates the population growth rate of the most populous countries over the years. The data label is disabled, with the tooltip displaying detailed data point values for each country and year combination.</p>
    
    <p><strong>Axis Inversion</strong> - This example demonstrates how to reverse the axis origin for both X and Y axes. Once the axis origin has been reversed, the axis data will be displayed in inverted order, effectively reversing the visualization orientation.</p>
    
    <p><strong>Configuration</strong> - You can reverse the axis origin by enabling the <a target='_blank' href='https://help.syncfusion.com/cr/blazor/Syncfusion.Blazor.HeatMap.HeatMapCommonAxis.html#Syncfusion_Blazor_HeatMap_HeatMapCommonAxis_IsInversed' aria-label="Class reference of IsInversed property in HeatMap Chart">IsInversed</a> property for each axis, providing flexibility in data presentation based on your analytical requirements.</p>
    
    <p><strong>Use Cases</strong> - Axis inversion is useful for:</p>
    <ul>
        <li>Displaying data in reverse chronological order</li>
        <li>Aligning with specific business logic or regional conventions</li>
        <li>Creating alternative data perspectives for comparative analysis</li>
    </ul>
    
    <p><strong>Interactive Features</strong> - The tooltip is enabled in this example. Hover the mouse over a data point or tap on touch-enabled devices to view population growth information with inverted axis ordering.</p>
</ActionDescription>

@code{
    public SfHeatMap<double[,]>? Map { get; set; }
    private Theme _theme { get; set; }
    public string AxisLabelColor = "Black";
    public bool XInverse = true;
    public bool YInverse = true;
    public string[] XLabels = new string[] { "China", "India", "USA", "Indonesia", "Brazil", "Pakistan", "Nigeria", "Bangladesh", "Russia", "Mexico" };
    public string[] YLabels = new string[] { "1965-1970", "1970-1975", "1975-1980", "1980-1985", "1985-1990", "1990-1995", "1995-2000", "2000-2005", "2005-2010", "2010-2015" };
    public double[,] DataSource = HeatMapData.GetInverseData();
    
    public void ValueXChange(Syncfusion.Blazor.Buttons.ChangeEventArgs<bool> args)
    {
        XInverse = args.Checked;
    }
    public void ValueYChange(Syncfusion.Blazor.Buttons.ChangeEventArgs<bool> args)
    {
        YInverse = args.Checked;
    }
    protected override void OnInitialized()
    {
        _theme = HeatmapThemeHelper.GetCurrentHeatmapTheme(NavigationManager.Uri);
        AxisLabelColor = (_theme.ToString().IndexOf("Dark") > -1 || _theme.ToString().IndexOf("HighContrast") > -1) ? "#FFFFFF" : "#000000";
    }
    /// <summary>
    /// EventCallback handler - invoked when PropertyPanel is expanded/collapsed
    /// This is triggered by CollapseChanged EventCallback from PropertyPanel
    /// </summary>
    public async Task HandlePanelCollapseChangedAsync()
    {

        if (Map != null)
        {
            await Map.RefreshBoundAsync();
        }
    }
}
