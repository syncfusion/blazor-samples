    <p>The File Upload component allows you to load an initial list of files that are already uploaded to the server. Pre-loaded files are useful for viewing and removing files from the server, providing users with visibility into existing uploads and management capabilities.</p>

@code {
    public SfUploader? UploaderObj;

    private void ClearFiles()
    {
        if (this.UploaderObj != null)
        {
            this.UploaderObj.ClearAllAsync();          
        }
    }
}
